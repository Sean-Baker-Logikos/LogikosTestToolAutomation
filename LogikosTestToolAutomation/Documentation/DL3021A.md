# DL3021A DC Electronic Load

# Example Usage

    from LogikosTestToolAutomation.DL3021A import DL3021A
    import time

    load = DL3021A()

    load.set_mode(DL3021A.Mode.FIX)
    load.set_function(DL3021A.Function.CURR)

    load.setup_const_current(0.5)
    print(load.query_const_current())

    load.on()

    time.sleep(5)
    voltage = load.get_voltage()
    current = load.get_current()
    print(f"{voltage}V {current}A")

    load.off()

# Documentation

### DL3021A Methods

    wait()
        Configures the instrument to wait for all pending operations to complete before executing any additional commands.

    on()
    off()

    set_mode(mode)
    get_mode()
        Sets or queries the input regulation mode.
        Note: Only DL3021A.Mode.FIX is currently supported by the LogikosTestToolAutomation code. Other modes can be added
              later if necessary.

            mode : See DL3021A.Mode

    set_function(fn : DL3021A.Function)
    get_function()
        Sets or queries the static operation mode.

            fn : See DL3021A.Function

    setup_const_current(current, range, slew, von, v_limit, c_limit)
    query_const_current() -> const_current_values
        Sets or queries the settings for constant current mode (DL3021A.Function.CURR)

            current : the load regulated current [Amps]
            range : the current range [Amps]
            slew : the rising and falling slew rate [Amps]
            von : the starting voltage [Volts]
            v_limit : the upper limit of the voltage [Volts]
            c_limit : the upper limit of the current [Amps]

            All parameters are optional, only specified parameters will be set. Parameters after current must be
            specified by name. For each parameter, "MIN", "MAX", or "DEF" may be passed to set the parameter to its
            minimum, maximum, or default value.

    setup_const_voltage(voltage, range, v_limit, c_limit)
    query_const_voltage() -> const_voltage_values
        Sets or queries the settings for constant voltage mode (DL3021A.Function.VOLT)

            voltage : the load voltage [Volts]
            range : the voltage range [Volts]
            v_limit : the upper limit of the voltage [Volts]
            c_limit : the upper limit of the current [Amps]

            All parameters are optional, only specified parameters will be set. Parameters after voltage must be
            specified by name. For each parameter, "MIN", "MAX", or "DEF" may be passed to set the parameter to its
            minimum, maximum, or default value.

    setup_const_resistance(resistance, range, v_limit, c_limit)
    query_const_resistance() -> const_resistance_values
        Sets or queries the settings for constant resistance mode (DL3021A.Function.RES)

            resistance : the load resistance [Ohms]
            range : the resistance range [Ohms]
            v_limit : the upper limit of the voltage [Volts]
            c_limit : the upper limit of the current [Amps]

            All parameters are optional, only specified parameters will be set. Parameters after resistance must be
            specified by name. For each parameter, "MIN", "MAX", or "DEF" may be passed to set the parameter to its
            minimum, maximum, or default value.

    setup_const_power(power, v_limit, c_limit)
    query_const_power() -> const_power_values
        Sets or queries the settings for constant power mode (DL3021A.Function.POW)

            power : the load power [Watts]
            v_limit : the upper limit of the voltage [Volts]
            c_limit : the upper limit of the current [Amps]

            All parameters are optional, only specified parameters will be set. Parameters after power must be
            specified by name. For each parameter, "MIN", "MAX", or "DEF" may be passed to set the parameter to its
            minimum, maximum, or default value.

### Data Types

    DL3021A.Mode
        FIX     = 0     # indicates that the input regulation mode is determined by the FUNC command.
        LIST    = 1     # indicates that the input regulation mode is determined by the activated list command.
        WAV     = 2     # indicates that the input regulation mode is determined by the waveform display command.
        BATT    = 3     # indicates that the input regulation mode is determined by the battery discharge command.
        OCP     = 4     # indicates that the input regulation mode is determined by the OCP command.
        OPP     = 5     # indicates that the input regulation mode is determined by the OPP command.

    DL3021A.Function
        CURR    = 0     # Constant current (CC) mode
        VOLT    = 1     # Constant voltage (CV) mode
        RES     = 2     # Constant resistance (CR) mode
        POW     = 3     # Constant power (CP) mode
